$Script:StartTime = Get-Date

$Script:TestsSkipped = 0
$Script:Results = @()

$BaseUrl = switch ($Environment) {

    Error   = 'Red'
    Warning = 'Yellow'
    Info    = 'Cyan'
    Dim     = 'Gray'

# ============================================================================
# 🛠️ Helper Functions
# ============================================================================

function Write-TestHeader {
    param([string]$Title)
    Write-Host "`n🔍 $Title" -ForegroundColor $Colors.Info
    Write-Host $('=' * 70) -ForegroundColor $Colors.Dim
}

function Write-TestResult {
    param(
        [string]$TestName,
        [bool]$Passed,
        [string]$Message = '',
        [string]$Category = 'general',
        [string]$Severity = 'normal'
    )
    
    $icon = if ($Passed) { '✅' } elseif ($Severity -eq 'critical') { '🔴' } else { '❌' }
    $color = if ($Passed) { $Colors.Success } elseif ($Severity -eq 'critical') { $Colors.Error } else { $Colors.Warning }
    
    Write-Host "  $icon $TestName" -ForegroundColor $color
    if ($Message) { Write-Host "     ↳ $Message" -ForegroundColor $Colors.Dim }
    
    $Script:Results += [PSCustomObject]@{
        Timestamp  = Get-Date -Format 'o'
        Category   = $Category
        Test       = $TestName
        Passed     = $Passed
        Message    = $Message
        Severity   = $Severity
    }
    
    if ($Passed) { $Script:TestsPassed++ } 
    elseif ($Severity -eq 'warning') { $Script:TestsSkipped++ }
    else { $Script:TestsFailed++ }
}

function Invoke-APITest {
    param(
        [string]$Endpoint,
        [string]$Method = 'Get',
        $Body = $null,
        $Headers = @{},
        [int]$TimeoutSec = 30,
        [string]$TestCategory = 'api'
    )
    
    try {
        $params = @{
            Uri = "$BaseUrl$Endpoint"
            Method = $Method
            TimeoutSec = $TimeoutSec
            ErrorAction = 'Stop'
        }
        
        if ($Body) {
            $params.ContentType = 'application/json'
            $params.Body = ($Body | ConvertTo-Json -Depth 10)
        }
        
        if ($Headers.Count -gt 0) {
            $params.Headers = $Headers
        }
        
        $response = Invoke-RestMethod @params
        return @{ Success = $true; Data = $response }
    }
    catch {
        return @{ Success = $false; Error = $_.Exception.Message }
    }
}

function Test-FileExists {
    param([string]$Path)
    $fullPath = Join-Path $Script:ProjectRoot $Path
    return Test-Path $fullPath
}

function Get-FileSize {
    param([string]$Path)
    $fullPath = Join-Path $Script:ProjectRoot $Path
    if (Test-Path $fullPath) {
        $size = (Get-Item $fullPath).Length
        return [Math]::Round($size / 1KB, 2)
    }
    return 0
}

function Test-ProjectStructure {
    Write-TestHeader "TEST 1: Project Structure & Files"

        "package.json",

    }
    
    # Key directories
    $directories = @(
        "src/app",
        "src/lib",
        "src/components",
        "src/hooks",
        "src/types",
        "prisma",
        "public"
    )
    
    foreach ($dir in $directories) {
        $exists = Test-Path (Join-Path $Script:ProjectRoot $dir)
        Write-TestResult "$dir directory" $exists $dir 'structure'
    }
    
    # Count total lines of code
    $tsFiles = Get-ChildItem -Path (Join-Path $Script:ProjectRoot "src") -Recurse -Include "*.ts","*.tsx"
    $totalLines = ($tsFiles | Get-Content | Measure-Object -Line).Lines
    Write-Host "  📊 Total TypeScript Lines: $totalLines" -ForegroundColor $Colors.Info
    Write-TestResult "Codebase size" ($totalLines -gt 50000) "$totalLines lines" 'structure'

function Test-MCPIntegrations {
    Write-TestHeader "TEST 2: MCP Integrations"
    
    # Check MCP files exist
    $mcpFiles = @(
        "src/app/api/mcp/available/route.ts",
        "src/app/api/mcp/call/route.ts",
        "src/app/api/mcp/functions/route.ts",
        "src/app/api/mcp-builder/load/route.ts",
        "src/app/api/mcp-builder/save/route.ts",
        "src/ai/context-manager.ts",
        "src/ai/mcp-monitor.ts",
        "src/ai/mcp-optimizer.ts"
    )
    
    foreach ($file in $mcpFiles) {
        $exists = Test-FileExists $file
        Write-TestResult "MCP: $file" $exists $file 'mcp'
    }
    
    # Test MCP API endpoint
    if ($TestCategory -contains 'ALL' -or $TestCategory -contains 'MCP') {
        $result = Invoke-APITest -Endpoint "/api/mcp/available" -TestCategory 'mcp'
        if ($result.Success) {
            Write-TestResult "MCP Available API" $true "$($result.Data.mcps.Count) MCPs available" 'mcp'
        } else {
            Write-TestResult "MCP Available API" $false $result.Error 'mcp' 'critical'
        }
    }
    
    # Check environment variables
    $envVars = @('GITHUB_TOKEN', 'FIRECRAWL_API_KEY', 'CONTEXT7_API_KEY', 'VERCEL_TOKEN')
    foreach ($var in $envVars) {
        $value = Get-Content (Join-Path $Script:ProjectRoot ".env") -ErrorAction SilentlyContinue | Where-Object {$_ -match "^$var="}
        $hasValue = $value -and ($value -split "=")[1] -ne ""
        Write-TestResult "Env: $var" $hasValue $var 'mcp'
    }
}

# 🧪 TEST 9: Dependencies & Package

function Test-Dependencies {
    Write-TestHeader "TEST 9: Dependencies & Package"
    
    # Check package.json
    $packageExists = Test-FileExists "package.json"
    Write-TestResult "package.json exists" $packageExists "package.json" 'dependencies'
    
    if ($packageExists) {
        $packageJson = Get-Content (Join-Path $Script:ProjectRoot "package.json") -Raw | ConvertFrom-Json
        
        # Count dependencies
        $depCount = $packageJson.dependencies.Count
        $devDepCount = $packageJson.devDependencies.Count
        Write-TestResult "Dependencies" ($depCount -gt 30) "$depCount deps, $devDepCount devDeps" 'dependencies'
        
        # Check critical dependencies
        $criticalDeps = @('next', 'react', 'prisma', '@prisma/client', 'socket.io', 'web-push')
        foreach ($dep in $criticalDeps) {
            $hasDep = $packageJson.dependencies.ContainsKey($dep) -or $packageJson.devDependencies.ContainsKey($dep)
            Write-TestResult "Critical dep: $dep" $hasDep $dep 'dependencies'
        }
    }
    
    # Check node_modules
    $nodeModulesExists = Test-Path (Join-Path $Script:ProjectRoot "node_modules")
    Write-TestResult "node_modules installed" $nodeModulesExists "Dependencies installed" 'dependencies' 'critical'
}

# 📊 Final Report
# ============================================================================

function Write-FinalReport {
    $duration = (New-TimeSpan $Script:StartTime (Get-Date)).TotalSeconds
    $total = $Script:TestsPassed + $Script:TestsFailed
    $passRate = if ($total -gt 0) { [Math]::Round(($Script:TestsPassed / $total) * 100, 1) } else { 0 }
    
    Write-Host "`n" + ('=' * 70) -ForegroundColor $Colors.Dim
    Write-Host "📊 AGENTFLOW PRO - COMPLETE TEST REPORT" -ForegroundColor $Colors.Info
    Write-Host $('=' * 70) -ForegroundColor $Colors.Dim
    Write-Host "  Environment: $Environment" -ForegroundColor $Colors.Dim
    Write-Host "  Duration: $([Math]::Round($duration, 1))s" -ForegroundColor $Colors.Dim
    Write-Host "`n  ✅ Passed: $Script:TestsPassed" -ForegroundColor $Colors.Success
    Write-Host "  ❌ Failed: $Script:TestsFailed" -ForegroundColor $Colors.Error
    Write-Host "  ⏭️  Skipped: $Script:TestsSkipped" -ForegroundColor $Colors.Warning
    Write-Host "  📈 Pass Rate: $passRate%" -ForegroundColor $(if($passRate -ge 90){$Colors.Success}elseif($passRate -ge 70){$Colors.Warning}else{$Colors.Error})
    
    # Category breakdown
    Write-Host "`n📋 Results by Category:" -ForegroundColor $Colors.Info
    $categories = $Script:Results | Group-Object Category
    foreach ($cat in $categories) {
        $passed = ($cat.Group | Where-Object { $_.Passed }).Count
        $total = $cat.Group.Count
        Write-Host "  $($cat.Name): $passed/$total" -ForegroundColor $(if($passed -eq $total){$Colors.Success}else{$Colors.Warning})
    }
    
    # Export if requested
    if ($ExportResults) {
        $timestamp = Get-Date -Format 'yyyyMMdd_HHmmss'
        $outputFile = "test-results-complete-$Environment-$timestamp.json"
        $Script:Results | ConvertTo-Json -Depth 10 | Out-File -FilePath $outputFile -Encoding utf8
        Write-Host "`n  💾 Results exported to: $outputFile" -ForegroundColor $Colors.Info
    }
    
    # Critical failures
    $criticalFailures = $Script:Results | Where-Object { -not $_.Passed -and $_.Severity -eq 'critical' }
    if ($criticalFailures) {
        Write-Host "`n🔴 CRITICAL FAILURES (must fix before launch):" -ForegroundColor $Colors.Error
        foreach ($failure in $criticalFailures) {
            Write-Host "  • $($failure.Test): $($failure.Message)" -ForegroundColor $Colors.Error
        }
    }
    
    # Final status
    Write-Host "`n" + ('=' * 70) -ForegroundColor $Colors.Dim
    if ($Script:TestsFailed -eq 0) {
        Write-Host "🎉 VSI TESTI USPEŠNI! AgentFlow Pro je ready za production." -ForegroundColor $Colors.Success
        Write-Host "`n🚀 Naslednji koraki:" -ForegroundColor $Colors.Info
        Write-Host "  1. npm run build" -ForegroundColor $Colors.Dim
        Write-Host "  2. vercel --prod" -ForegroundColor $Colors.Dim
        Write-Host "  3. Monitoriraj Sentry dashboard" -ForegroundColor $Colors.Dim
        return 0
    } else {
        Write-Host "⚠️  Nekateri testi so spodleteli. Popravi napake pred launchom." -ForegroundColor $Colors.Warning
        Write-Host "`n🔧 Priporočila:" -ForegroundColor $Colors.Info
        Write-Host "  1. Preveri .env konfiguracijo" -ForegroundColor $Colors.Dim
        Write-Host "  2. Zaženi npm install" -ForegroundColor $Colors.Dim
        Write-Host "  3. Preveri API endpoint-e" -ForegroundColor $Colors.Dim
        return 1
    }
}
