<#
.SYNOPSIS
    AgentFlow Pro - PWA Testing Suite
    Testira manifest, ikone, meta tage in live server za mobile app functionality

.USAGE
    .\scripts\test-pwa.ps1
    .\scripts\test-pwa.ps1 -OpenBrowser
    .\scripts\test-pwa.ps1 -BaseUrl "http://localhost:3002"
#>

[CmdletBinding()]
param(
    [string]$BaseUrl = "http://localhost:3002",
    [switch]$OpenBrowser,
    [switch]$ExportResults
)

# ============================================================================
# 🎨 Konfiguracija
# ============================================================================

$ProjectRoot = Split-Path -Parent $PSScriptRoot
$Results = @()
$Passed = 0
$Failed = 0

$Colors = @{ Success='Green'; Error='Red'; Warning='Yellow'; Info='Cyan'; Dim='Gray' }

function Write-Test {
    param([string]$Name, [bool]$Ok, [string]$Msg = '')
    $icon = if($Ok){'✅'}else{'❌'}; $color = if($Ok){$Colors.Success}else{$Colors.Error}
    Write-Host "  $icon $Name" -ForegroundColor $color
    if($Msg){ Write-Host "     ↳ $Msg" -ForegroundColor $Colors.Dim }
    $script:Results += @{Test=$Name; Passed=$Ok; Message=$Msg}
    if($Ok){ $script:Passed++ } else { $script:Failed++ }
}

function Test-File { param([string]$Path); Test-Path (Join-Path $ProjectRoot $Path) }
function Test-Contains { 
    param([string]$Path, [string]$Pattern)
    $full = Join-Path $ProjectRoot $Path
    if(!(Test-Path $full)){ return $false }
    (Get-Content $full -Raw) -match $Pattern
}

# ============================================================================
# 🧪 TESTI
# ============================================================================

Write-Host "`n🤖 AgentFlow Pro – PWA Test Suite" -ForegroundColor Cyan
Write-Host "   Base: $BaseUrl | Port: 3002" -ForegroundColor Gray
Write-Host $('='*60)

# 1. Manifest
Write-Host "`n📄 Manifest File" -ForegroundColor Cyan
$manifestExists = Test-File "public/manifest.json"
Write-Test "manifest.json exists" $manifestExists "public/manifest.json"

if($manifestExists){
    $manifest = Get-Content (Join-Path $ProjectRoot "public/manifest.json") -Raw | ConvertFrom-Json -ErrorAction SilentlyContinue
    Write-Test "Valid JSON" ($null -ne $manifest) "Parsing OK"
    if($manifest){
        Write-Test "Has 'name'" ($null -ne $manifest.name) "$($manifest.name)"
        Write-Test "Has 'start_url'" ($manifest.start_url -eq '/') "Value: $($manifest.start_url)"
        Write-Test "Display 'standalone'" ($manifest.display -eq 'standalone') "Value: $($manifest.display)"
        Write-Test "Has icons" ($manifest.icons.Count -gt 0) "Count: $($manifest.icons.Count)"
    }
} else {
    Write-Host "  ⚠️  Ustvarjam sample manifest.json..." -ForegroundColor Yellow
    $sample = @{
        name="AgentFlow Pro"; short_name="AgentFlow"; start_url="/"; display="standalone"
        background_color="#ffffff"; theme_color="#2563eb"
        icons=@(@{src="/icons/icon-192x192.png";sizes="192x192";type="image/png"},
                @{src="/icons/icon-512x512.png";sizes="512x512";type="image/png"})
    }
    $sample | ConvertTo-Json -Depth 10 | Out-File (Join-Path $ProjectRoot "public/manifest.json") -Encoding utf8
    Write-Host "  ✅ Created public/manifest.json" -ForegroundColor Green
}

# 2. Ikone
Write-Host "`n🖼️  PWA Icons" -ForegroundColor Cyan
Write-Test "icons/ folder exists" (Test-Path (Join-Path $ProjectRoot "public/icons")) "public/icons/"
Write-Test "icon-192x192.png" (Test-File "public/icons/icon-192x192.png") "Required for Android"
Write-Test "icon-512x512.png" (Test-File "public/icons/icon-512x512.png") "Required for PWA"
if(!(Test-File "public/icons/icon-192x192.png")){
    Write-Host "  💡 Generiraj ikone: https://realfavicongenerator.net" -ForegroundColor Cyan
}

# 3. Meta Tags v layout.tsx
Write-Host "`n🏷️  Meta Tags (src/app/layout.tsx)" -ForegroundColor Cyan
$layout = "src/app/layout.tsx"
Write-Test "layout.tsx exists" (Test-File $layout) $layout
if(Test-File $layout){
    Write-Test "manifest link" (Test-Contains $layout 'rel="manifest"") '<link rel="manifest"'
    Write-Test "apple-touch-icon" (Test-Contains $layout 'apple-touch-icon') 'apple-touch-icon'
    Write-Test "theme-color" (Test-Contains $layout 'theme-color') 'theme-color="#2563eb"'
    Write-Test "apple-mobile-web-app-capable" (Test-Contains $layout 'apple-mobile-web-app-capable') 'iOS fullscreen'
    Write-Test "mobile-web-app-capable" (Test-Contains $layout 'mobile-web-app-capable') 'Android fullscreen'
}

# 4. Live Server Testi
Write-Host "`n🌐 Live Server ($BaseUrl)" -ForegroundColor Cyan
try {
    $resp = Invoke-WebRequest -Uri $BaseUrl -TimeoutSec 10 -ErrorAction Stop
    Write-Test "Server running" $true "HTTP $($resp.StatusCode)"
    
    # Manifest accessible
    $manifestResp = Invoke-WebRequest -Uri "$BaseUrl/manifest.json" -TimeoutSec 10 -ErrorAction SilentlyContinue
    Write-Test "manifest.json accessible" ($manifestResp.StatusCode -eq 200) "HTTP $($manifestResp.StatusCode)"
    
    # Check HTML for manifest link
    Write-Test "manifest link in HTML" ($resp.Content -match 'rel="manifest"') '<link rel="manifest">'
} catch {
    Write-Test "Server running" $false "Start with: npm run dev"
    Write-Host "  💡 Run 'npm run dev' first (port 3002)" -ForegroundColor Yellow
}

# 5. Lighthouse Guide
Write-Host "`n🔦 Lighthouse PWA Audit (Manual)" -ForegroundColor Cyan
Write-Host "  1. Chrome → $BaseUrl" -ForegroundColor Gray
Write-Host "  2. F12 → Lighthouse tab" -ForegroundColor Gray  
Write-Host "  3. Select 'Progressive Web App'" -ForegroundColor Gray
Write-Host "  4. Click 'Analyze page load'" -ForegroundColor Gray
Write-Host "  5. Target: PWA score 100/100 ✅" -ForegroundColor Gray

# 6. Mobile Testing Guide
Write-Host "`n📱 Mobile Testing (Manual)" -ForegroundColor Cyan
Write-Host "  iPhone (Safari):" -ForegroundColor Gray
Write-Host "    • Share → Add to Home Screen → Add" -ForegroundColor Gray
Write-Host "    • Preveri: fullscreen, brez Safari vrstice" -ForegroundColor Gray
Write-Host "  Android (Chrome):" -ForegroundColor Gray  
Write-Host "    • ⋮ menu → Add to Home screen → Install" -ForegroundColor Gray
Write-Host "    • Preveri: fullscreen, brez Chrome vrstice" -ForegroundColor Gray

# ============================================================================
# 📊 Report
# ============================================================================

$Total = $Passed + $Failed
$Rate = if($Total -gt 0){ [Math]::Round(($Passed/$Total)*100,1) }else{ 0 }

Write-Host "`n" + ('='*60) -ForegroundColor Gray
Write-Host "📊 PWA TEST REPORT" -ForegroundColor Cyan
Write-Host "  ✅ Passed: $Passed" -ForegroundColor Green
Write-Host "  ❌ Failed: $Failed" -ForegroundColor Red  
Write-Host "  📈 Pass Rate: $Rate%" -ForegroundColor $(if($Rate -ge 90){'Green'}elseif($Rate -ge 70){'Yellow'}else{'Red'})

if($ExportResults){
    $file = "pwa-results-$(Get-Date -Format 'yyyyMMdd_HHmmss').json"
    $Results | ConvertTo-Json | Out-File $file
    Write-Host "  💾 Exported: $file" -ForegroundColor Cyan
}

if($OpenBrowser){ Start-Process $BaseUrl }

Write-Host "`n" + ('='*60) -ForegroundColor Gray
if($Failed -eq 0){
    Write-Host "🎉 VSI TESTI USPEŠNI! PWA ready za production." -ForegroundColor Green
} else {
    Write-Host "⚠️  Popravi ❌ napake pred launchom." -ForegroundColor Yellow
}
